Example Queries
  • Basic:
  • SELECT * FROM range(10)
  • SELECT i, i*2 AS doubled FROM range(5) t(i)
  • SELECT 'Hello' || ' ' || 'World' AS greeting
  • SELECT current_timestamp AS now, version() AS version
  • CREATE TABLE test (id INT, name VARCHAR); INSERT INTO test VALUES (1, 'Alice'), (2, 'Bob'); SELECT * FROM test;
  • JSON:
  • SELECT json('{"name": "Alice", "age": 30}')
  • SELECT json_extract_string('{"user": "Bob"}', '$.user') AS user
  • SELECT '{"id": 1, "active": true}'::JSON->>'$.active' AS active
  • SELECT json_keys('{"a": 1, "b": 2, "c": 3}') AS keys
  • Remote (httpfs):
  • SELECT * FROM 'https://shell.duckdb.org/data/tpch/0_01/parquet/orders.parquet' LIMIT 5
